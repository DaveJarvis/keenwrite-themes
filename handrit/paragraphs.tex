% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% 
%
% Configure paragraphs.
%
% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% 

\setupinterlinespace[line=24pt]

\setupindenting[.5in, yes, first]
\setupheads[indentnext=yes]

\setupalign[
  hz,
  hanging,
  nothyphenated,
  extremestretch,
  tolerant,
  flushleft,
]

% Equate \it with \em
\setupbodyfontenvironment[default][em=italic]

% Allow line-breaking at naturally-hyphenated words.
\setbreakpoints[compound]

\setuptolerance[
  extremestretch,
  verytolerant,
]
