% Configure widows and orphans.
%
% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% 

% Prevent widows and orphans by forcing lines together.
\startsetups[grid][TextPenalties]
  \setdefaultpenalties
  \setpenalties\widowpenalties{2}{10000}
  \setpenalties\clubpenalties {2}{10000}
\stopsetups

\setuplayout[
  setups=TextPenalties,
]
