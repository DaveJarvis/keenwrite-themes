% Configure title page.
%

\startsetups document:start
  \startstandardmakeup
    \startalignment[middle]
      {\ss\documentvariable{title}}
      \blank[medium]
      \documentvariable{byline}
    \stopalignment
  \stopstandardmakeup
\stopsetups
