% Configure TODO texts.

% Allow TODO to span pages.
\definetextbackground[TextTodo][
  location=paragraph,
  background=off,
  framecolor=darkred,
  corner=round,
  rulethickness=1pt,
  before=\blank,
  after=\blank,
  topoffset=.5em,
  leftoffset=1.5em,
  rightoffset=1.5em,
]

% Create an unmissable box with TODO at the top.
\definestartstop[todo][
  before={\startTextTodo
    \startalignment[middle]
      \bf{\color[red]{TODO}}
    \stopalignment
    \blank[disable]
  },
  after={\stopTextTodo},
]
