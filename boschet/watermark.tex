\definecolor[TextWatermarkColour][h=D8D8D8]

% Reference document variables through a setups environment.
\startsetups TextReviewerName 
% Use a frame to center vertically.
\framed[frame=off]{%
  \doifdocumentvariable{reviewer}{
    \color[TextWatermarkColour]{%
      \it
      \rotate[rotation=-90]{Alpha copy for \documentvariable{reviewer}}
    }
  }
}
\stopsetups

\defineoverlay[TextWatermarkOverlay][\setups{TextReviewerName}]
\setupbackgrounds[text][rightmargin][background=TextWatermarkOverlay]
