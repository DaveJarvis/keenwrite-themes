\definelines[TextPoemLyrics]
\setuplines[TextPoemLyrics][indenting={even}]

\startxmlsetups xml:lyrics
  \startTextPoemLyrics
    \startnarrower[.5in]
      % Respect newlines.
      \em{\xmlflushlinewise{#1}}
    \stopnarrower
  \stopTextPoemLyrics

  \indentation
\stopxmlsetups
