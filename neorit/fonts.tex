% Configure font names and sizes.

\definefontfeature[TextFontFeature][default][
  liga=yes,
  tlig=yes,
  mode=node,
  protrusion=quality,
  expansion=quality
]

\definefontfamily [TextFont] [rm] [Tex Gyre Schola] [features=TextFontFeature]
\definefontfamily [TextFont] [ss] [Roboto]

\setupbodyfont[TextFont, 12pt]
