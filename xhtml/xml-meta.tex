% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% 
%
% Map metadata (head) elements to front matter macros
%
% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% 

\startxmlsetups xml:meta
  \startfrontmatter
    \xmlflush{#1}
    \placecontent
  \stopfrontmatter
\stopxmlsetups

\startxmlsetups xml:title
  \startstandardmakeup
    \title{\xmlflush{#1}}
  \stopstandardmakeup
\stopxmlsetups
